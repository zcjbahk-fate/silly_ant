  <script type="module">
    /* 辅助函数：安全获取嵌套对象的值，严格使用 lodash 的 get 方法 */
    function safeGet(obj, path, defaultValue = '未知') {
      return _.get(obj, path, defaultValue);
    }

    /* 渲染主角与系统状态 */
    function renderPlayerAndSystem(all_variables) {
      const daysLeft = safeGet(all_variables, 'stat_data.系统.距离除夕还剩天数', 7);
      $('#sys-countdown').text(daysLeft);

      /* 主角基础信息 */
      $('#player-age').text(safeGet(all_variables, 'stat_data.主角.年龄', 24));

      /* 肉棒状态 */
      $('#player-dick-len').text(safeGet(all_variables, 'stat_data.主角.性器状态.肉棒.当前长度', 0) + ' cm');
      $('#player-dick-state').text(safeGet(all_variables, 'stat_data.主角.性器状态.肉棒.状态', '未勃起'));
      $('#player-dick-desc').text(safeGet(all_variables, 'stat_data.主角.性器状态.肉棒.描述', '无'));

      /* 主角性经历记录表 */
      const expData = safeGet(all_variables, 'stat_data.主角.性经历', {});
      const $expTbody = $('#player-exp-tbody');
      $expTbody.empty();

      if (Object.keys(expData).length === 0) {
        $expTbody.append('<tr><td colspan="6">暂无配种记录</td></tr>');
      } else {
        Object.entries(expData).forEach(([name, data]) => {
          const k = safeGet(data, '口交', 0);
          const z = safeGet(data, '足交', 0);
          const c = safeGet(data, '插入', 0);
          const g = safeGet(data, '肛交', 0);
          const s = safeGet(data, '手交', 0);
          const total = safeGet(data, '总数', (k+z+c+g+s));
          $expTbody.append(`
            <tr>
              <td>${name}</td><td>${k}</td><td>${z}</td><td>${c}</td><td>${g}</td><td>${total}</td>
            </tr>
          `);
        });
      }
    }

    /* 渲染NPC列表 */
    function renderNpcs(all_variables) {
      const npcs = safeGet(all_variables, 'stat_data.角色', {});
      const $container = $('#npc-container');
      $container.empty();

      Object.entries(npcs).forEach(([name, data]) => {
        /* 为了在循环中方便，我们拼装完整的基础路径 */
        const basePath = `stat_data.角色.${name}`;

        const age = safeGet(all_variables, `${basePath}.年龄`, '未知');
        const known = safeGet(all_variables, `${basePath}.是否知晓村中习俗`, '未知');

        const love = safeGet(all_variables, `${basePath}.好感度`, 0);
        const corr = safeGet(all_variables, `${basePath}.堕落度`, 0);
        const lust = safeGet(all_variables, `${basePath}.性欲值`, 0);

        const posture = safeGet(all_variables, `${basePath}.当前姿势`, '未知');
        const mind = safeGet(all_variables, `${basePath}.内心想法`, '无');
        const view = safeGet(all_variables, `${basePath}.对主角看法`, '无');

        /* 解析精液状态 */
        const semenObj = safeGet(all_variables, `${basePath}.体内精液`, {});
        const semenList = Object.entries(semenObj).map(([src, info]) => {
          return `[${src}]: ${safeGet(info, '量', 0)}ml (${safeGet(info, '状态', '')})`;
        });
        const semenText = semenList.length > 0 ? semenList.join('<br>') : '干净';

        /* 上次做爱 */
        const lsTime = safeGet(all_variables, `${basePath}.上次做爱.时间`, '无');
        const lsObj = safeGet(all_variables, `${basePath}.上次做爱.对象`, '无');
        const lsLoc = safeGet(all_variables, `${basePath}.上次做爱.地点`, '无');
        const lsWay = safeGet(all_variables, `${basePath}.上次做爱.方式`, '无');

        /* NTR 与 出轨 */
        const kuzhu = safeGet(all_variables, `${basePath}.NTR关系.苦主姓名`, '无');
        const chuGui = safeGet(all_variables, `${basePath}.NTR关系.出轨记录`, {});
        let chuGuiHtml = '';
        if (Object.keys(chuGui).length > 0) {
          chuGuiHtml = Object.entries(chuGui).map(([hm, info]) =>
            `[${hm}] 侵犯度: ${safeGet(info, 'NTR', 0)}/100`
          ).join('<br>');
        } else {
          chuGuiHtml = '暂无越界记录';
        }

        /* NPC 性经历 */
        const expData = safeGet(all_variables, `${basePath}.性经历`, {});
        let expRows = '';
        if (Object.keys(expData).length === 0) {
          expRows = '<tr><td colspan="6">尚待开发</td></tr>';
        } else {
          Object.entries(expData).forEach(([objName, objData]) => {
            const k = safeGet(objData, '口交', 0);
            const z = safeGet(objData, '足交', 0);
            const c = safeGet(objData, '插入', 0);
            const g = safeGet(objData, '肛交', 0);
            const s = safeGet(objData, '手交', 0);
            const total = safeGet(objData, '总数', (k+z+c+g+s));
            expRows += `<tr><td>${objName}</td><td>${k}</td><td>${z}</td><td>${c}</td><td>${g}</td><td>${total}</td></tr>`;
          });
        }

        /* 构建单人卡片 HTML */
        const html = `
          <div class="npc-card">
            <div class="npc-header">
              <span class="npc-name">${name}</span>
              <span class="npc-basic-info">${age}岁 | 认知: <span class="${known === '已知' ? 'highlight-red' : ''}">${known}</span></span>
            </div>

            <div class="npc-bars">
              <div class="bar-container">
                <div class="bar-fill love" style="width: ${love}%;"></div>
                <div class="bar-text">好感 ${love}/100</div>
              </div>
              <div class="bar-container">
                <div class="bar-fill corr" style="width: ${corr}%;"></div>
                <div class="bar-text">堕落 ${corr}/100</div>
              </div>
              <div class="bar-container">
                <div class="bar-fill lust" style="width: ${lust}%;"></div>
                <div class="bar-text">发情 ${lust}/100</div>
              </div>
            </div>

            <!-- 折叠面板 1：当前状态 -->
            <button class="accordion-btn">实时监控</button>
            <div class="accordion-content">
              <div class="detail-item"><div class="stat-label">姿势躯干:</div><div class="stat-value">${posture}</div></div>
              <div class="detail-item"><div class="stat-label">真实心声:</div><div class="text-italic">"${mind}"</div></div>
              <div class="detail-item"><div class="stat-label">对你评估:</div><div class="stat-value">${view}</div></div>
            </div>

            <!-- 折叠面板 2：身体使用记录 -->
            <button class="accordion-btn">使用日志</button>
            <div class="accordion-content">
              <div class="detail-item">
                <div class="stat-label">内腔储精池:</div>
                <div class="highlight-red" style="text-align: right; margin-top: 4px;">${semenText}</div>
              </div>
              <div class="detail-item">
                <div class="stat-label">上次处理:</div>
                <div style="text-align: right; color:#b0a0a0; font-size: 0.9em; margin-top:4px;">
                  时间: ${lsTime} | 对象: ${lsObj}<br>地点: ${lsLoc} | 方式: ${lsWay}
                </div>
              </div>
              <div class="detail-item">
                <div class="stat-label">苦主 [${kuzhu}] 的绿帽档案:</div>
                <div style="text-align: right; color:#b0a0a0; font-size: 0.9em; margin-top:4px;">${chuGuiHtml}</div>
              </div>
            </div>

            <!-- 折叠面板 3：详细性经历表格 -->
            <button class="accordion-btn">受孕统计表</button>
            <div class="accordion-content" style="padding: 4px;">
              <table class="record-table">
                <thead><tr><th>对象</th><th>口</th><th>足</th><th>插</th><th>肛</th><th>总</th></tr></thead>
                <tbody>${expRows}</tbody>
              </table>
            </div>
          </div>
        `;
        $container.append(html);
      });
    }

    /* 绑定UI交互事件 */
    function bindEvents() {
      /* Tab 切换 */
      $('.nav-btn').off('click').on('click', function() {
        $('.nav-btn').removeClass('active');
        $(this).addClass('active');
        const target = $(this).data('target');
        $('.tab-content').removeClass('active');
        $('#' + target).addClass('active');
      });

      /* 折叠面板展开/收起（使用事件委托机制处理动态生成的元素） */
      $(document).off('click', '.accordion-btn').on('click', '.accordion-btn', function() {
        $(this).toggleClass('active');
        $(this).next('.accordion-content').slideToggle(200);
      });
    }

    /* 主更新逻辑 */
    function populateCharacterData() {
      const all_variables = getAllVariables();

      renderPlayerAndSystem(all_variables);
      renderNpcs(all_variables);

      /* 重新绑定事件以确保动态元素可点击 */
      bindEvents();
    }

    /* 初始化 */
    async function init() {
      await waitGlobalInitialized('Mvu');
      populateCharacterData();

      /* 监听变量更新事件，实现自动刷新 */
      eventOn(Mvu.events.VARIABLE_UPDATE_ENDED, () => {
        populateCharacterData();
      });
    }

    /* 启动入口 */
    $(errorCatched(init));
  </script>
