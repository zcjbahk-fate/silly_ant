            --novel-text: #e0e5f0;
            --novel-card-bg: rgba(255, 255, 255, 0.08); /* 提亮卡片背景 */
            --novel-border: rgba(173, 216, 230, 0.25);  /* 冰蓝边框 */
            --novel-tag-bg: rgba(0, 0, 0, 0.2);
            /* 强制覆盖子元素文字颜色 */

        }

        /* 黑夜模式下的标题颜色强制变亮 */
        .novel-header-container.night-mode .novel-title {

        /* 开关按钮 */

    <!-- 切换按钮：默认显示太阳 -->

    <span class="novel-chapter-num">$1</span>
    <div class="novel-title">$2</div>
    <div id="novel-meta-container" data-cp="$3" data-loc="$4"></div>
    <div class="novel-tags-wrapper" id="novel-tags-container" data-tags="$5"></div>
    <div class="novel-preface">$6</div>

    <script>
        (function () {
            // === 1. 初始化日夜模式 (记忆功能核心) ===
            // 获取当前卡片的容器和按钮
            // 注意：因为这段脚本在每个卡片加载时都会运行，我们使用 currentScript 可以精确定位当前容器
            const currentScript = document.currentScript;
            const container = currentScript.closest('.novel-header-container');
            const btn = container.querySelector('.theme-toggle-btn');
            const icon = btn.querySelector('.toggle-icon');

            // 读取全局偏好
            const savedPref = localStorage.getItem('novel_theme_preference');

            // 如果记忆里是黑夜，直接应用，不闪烁
            if (savedPref === 'night') {
                container.classList.add('night-mode');
                icon.textContent = '☽';
            }

            // === 2. 绑定切换功能 ===
            // 挂载到按钮上，通过闭包直接引用 container，避免全局污染
            btn.onclick = function() {
                container.classList.toggle('night-mode');

                const isNight = container.classList.contains('night-mode');
                icon.textContent = isNight ? '☽' : '☀';

                // 保存偏好到本地存储
                localStorage.setItem('novel_theme_preference', isNight ? 'night' : 'day');
            };

            // === 3. 原有的渲染逻辑 ===
            const metaContainer = container.querySelector('#novel-meta-container');
            const cpRaw = metaContainer.getAttribute('data-cp') || '';
            const locRaw = metaContainer.getAttribute('data-loc') || '';

            const cpGroups = cpRaw.split(/[,，]/).map(g => g.trim()).filter(g => g);
            const isSingleCP = cpGroups.length <= 1;

            function parseCPGroup(group) {
                const parts = group.split(/\s*([×xX&])\s*/);
                let html = '';
                parts.forEach((part, idx) => {
                    if (!part.trim()) return;
                    if (idx % 2 === 1) {
                        html += '<span class="novel-cp-relation">' + part.trim() + '</span>';
                    } else {
                        html += '<span class="novel-cp-name">' + part.trim() + '</span>';
                    }
                });
                return html;
            }

            if (isSingleCP) {
                metaContainer.innerHTML = `
                    <div class="novel-meta-single-row">
                        <div class="novel-meta-card">
                            <span class="novel-meta-label">CP</span>
                            <div class="novel-cp-inline">${cpGroups[0] ? parseCPGroup(cpGroups[0]) : '-'}</div>
                        </div>
                        <div class="novel-meta-card">
                            <span class="novel-meta-label">AT</span>
                            <span>${locRaw}</span>
                        </div>
                    </div>`;
            } else {
                let cpGroupsHTML = cpGroups.map(g => `<div class="novel-cp-group">${parseCPGroup(g)}</div>`).join('');
                metaContainer.innerHTML = `
                    <div class="novel-meta-multi-row">
                        <div class="novel-cp-row">
                            <span class="novel-cp-row-label">CP</span>
                            ${cpGroupsHTML}
                        </div>
                        <div class="novel-meta-card">
                            <span class="novel-meta-label">AT</span>
                            <span>${locRaw}</span>
                        </div>
                    </div>`;
            }

            const tagsContainer = container.querySelector('#novel-tags-container');
            if (tagsContainer) {
                const tagsRaw = tagsContainer.getAttribute('data-tags') || '';
                const tags = tagsRaw.split(/[\/，,]/).map(t => t.trim()).filter(t => t);
                tagsContainer.innerHTML = '';
                tags.forEach(tag => {
                    const span = document.createElement('span');
                    span.className = 'novel-tag';
                    span.textContent = tag;
                    tagsContainer.appendChild(span);
                });
            }
        })();
    </script>
</div>
</body>
</html>
