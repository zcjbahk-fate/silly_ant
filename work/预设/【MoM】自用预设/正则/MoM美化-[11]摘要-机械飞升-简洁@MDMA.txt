<script>
(function() {
    window.toggleGospelTheme = function(event) {
        if (event) event.stopPropagation();
        const root = document.documentElement;
        const currentTheme = root.getAttribute('data-gospel-theme') || 'day';
        const newTheme = currentTheme === 'day' ? 'night' : 'day';
        root.setAttribute('data-gospel-theme', newTheme);
        localStorage.setItem('gospel-theme', newTheme);
    };

    window.toggleGospelCard = function(headerEl) {
        if (!headerEl) return;
        const card = headerEl.closest('.gospel-card');
        if (card) card.classList.toggle('is-expanded');
    };

    window.toggleGospelSpoiler = function(btnEl) {
        if (!btnEl) return;
        const spoilerContainer = btnEl.closest('.gospel-spoiler');
        if (spoilerContainer && !spoilerContainer.classList.contains('is-empty')) {
            spoilerContainer.classList.toggle('is-revealed');
        }
    };

    setTimeout(() => {
        try {
            const savedTheme = localStorage.getItem('gospel-theme') || 'day';
            document.documentElement.setAttribute('data-gospel-theme', savedTheme);

            document.querySelectorAll('.gospel-data-5').forEach(el => {
                const container = el.closest('.gospel-spoiler');
                if (!container) return;
                
                let text = el.innerText || el.textContent || '';
                text = text.replace(/[\s\n\r]/g, '');
                
                const placeholder = '$' + '5'; 
                
                if (!text || text === placeholder) {
                    container.classList.add('is-empty');
                } else {
                    container.classList.remove('is-empty');
                }
            });
        } catch (error) {
            console.error("Gospel StatusBar Error:", error);
        }
    }, 150);
})();
</script>
