        // --- 解析XML格式的语录 ---
        const parseQuotes = (xmlString) => {
            if (!xmlString || xmlString.startsWith('$')) return [];
            const regex = /<语录>(.*?)<\/语录>/gs;
            const matches = [];
            let match;
            while ((match = regex.exec(xmlString)) !== null) {
                matches.push(match[1].trim());
            }
            return matches;
        };

        // --- 解析XML格式的搜索记录 ---
        const parseSearchHistory = (xmlString) => {
            if (!xmlString || xmlString.startsWith('$')) return [];
            const recordRegex = /<搜索记录>(.*?)<\/搜索记录>/gs;
            const records = [];
            let recordMatch;

            while ((recordMatch = recordRegex.exec(xmlString)) !== null) {
                const recordContent = recordMatch[1];
                const contentMatch = /<内容>(.*?)<\/内容>/.exec(recordContent);
                const timeMatch = /<时间>(.*?)<\/时间>/.exec(recordContent);

                if (contentMatch && timeMatch) {
                    records.push({
                        content: contentMatch[1].trim(),
                        time: timeMatch[1].trim()
                    });
                }
            }
            return records;
        };

        // DIALOGUE DATA - 从语录XML解析
        const CHAR_DIALOGUES = parseQuotes(`$12`);

                                aria-label={isPlaying ? "暂停" : "播放"}
                            >
                                {isPlaying ? <PauseIcon /> : <PlayIcon />}
                            </button>
                        </div>
                    </div>
                </div>
            );
        };

        /* --- MAIN COMPONENT --- */
        const SwissDynamicIsland = () => {
            const [isExpanded, setIsExpanded] = useState(false);
            const [activeId, setActiveId] = useState(null);
            const [quoteIndex, setQuoteIndex] = useState(0);


            const [charName, setCharName] = useState(() => {
                try {
                    const savedName = localStorage.getItem('dsi_name');
                    return savedName || DEFAULT_NAME;
                } catch (e) {
                    return DEFAULT_NAME;
                }
            });

            const [bgUrl, setBgUrl] = useState(() => {
                try {
                    const savedBg = localStorage.getItem('dsi_bg');
                    return savedBg || DEFAULT_BG;
                } catch (e) {
                    return DEFAULT_BG;
                }
            });

            const [bgSaveStatus, setBgSaveStatus] = useState('');

            useEffect(() => {
                try {
                    localStorage.setItem('dsi_name', charName);
                } catch (e) {
                    console.warn('Failed to save name to localStorage:', e);
                }
            }, [charName]);

            useEffect(() => {
                if (bgUrl === DEFAULT_BG || bgUrl.startsWith('http')) {
                    try {
                        localStorage.setItem('dsi_bg', bgUrl);
                        setBgSaveStatus('saved');
                    } catch (e) {
                        console.warn('Failed to save bg to localStorage:', e);
                        setBgSaveStatus('session-only');
                    }
                }
            }, [bgUrl]);

            const [bubbles, setBubbles] = useState([]);

            const audioRef = useRef(null);
            const [songData, setSongData] = useState(null);
            const [isLoadingMusic, setIsLoadingMusic] = useState(false);
            const [isPlaying, setIsPlaying] = useState(false);
            const [progress, setProgress] = useState(0);
            const [duration, setDuration] = useState(0);
            const [volume, setVolume] = useState(0.8);
            const [lastSearchedSong, setLastSearchedSong] = useState('');

            const searchAndLoadSong = async (songName) => {
                if (!songName || songName === lastSearchedSong || songName.startsWith('$')) return;

                setIsLoadingMusic(true);
                setLastSearchedSong(songName);

                try {
                    if (typeof Music !== 'undefined' && Music.SearchMusic) {
                        const result = await Music.SearchMusic(songName);
                        if (result && result.Url) {
                            setSongData(result);
                            if (audioRef.current) {
                                audioRef.current.src = result.Url;
                                audioRef.current.load();
                            }
                        } else {
                            setSongData(null);
                        }
                    } else {
                        console.warn('Music API not loaded');
                        setSongData(null);
                    }
                } catch (error) {
                    console.error('Failed to search music:', error);
                    setSongData(null);
                } finally {
                    setIsLoadingMusic(false);
                }
            };

            useEffect(() => {
                const songName = SONG_NAME_FROM_STATUS;
                if (songName && !songName.startsWith('$')) {
                    searchAndLoadSong(songName);
                }
            }, []);

            useEffect(() => {
                const audio = audioRef.current;
                if (!audio) return;
                audio.volume = volume;

                const updateProgress = () => setProgress(audio.currentTime);
                const setAudioDuration = () => setDuration(audio.duration);
                const handleEnd = () => setIsPlaying(false);
                const handleCanPlay = () => setIsLoadingMusic(false);

                audio.addEventListener('timeupdate', updateProgress);
                audio.addEventListener('loadedmetadata', setAudioDuration);
                audio.addEventListener('ended', handleEnd);
                audio.addEventListener('canplay', handleCanPlay);

                return () => {
                    audio.removeEventListener('timeupdate', updateProgress);
                    audio.removeEventListener('loadedmetadata', setAudioDuration);
                    audio.removeEventListener('ended', handleEnd);
                    audio.removeEventListener('canplay', handleCanPlay);
                };
            }, [volume]);

            const togglePlay = (e) => {
                if(e) e.stopPropagation();
                const audio = audioRef.current;
                if (!audio || !songData?.Url) return;
                if (isPlaying) audio.pause();
                else audio.play();
                setIsPlaying(!isPlaying);
            };

            const handleSeek = (e) => {
                const audio = audioRef.current;
                if (!audio) return;
                const val = Number(e.target.value);
                audio.currentTime = val;
                setProgress(val);
            };

            const handleVolume = (e) => {
                const newVol = Number(e.target.value);
                if (audioRef.current) audioRef.current.volume = newVol;
                setVolume(newVol);
            };

            const handleMouseMove = (e) => {
                if (isExpanded) return;
                const rect = e.currentTarget.getBoundingClientRect();
                const x = e.clientX - rect.left;
                const y = e.clientY - rect.top;
                e.currentTarget.style.setProperty('--mouse-x', `${x}px`);
                e.currentTarget.style.setProperty('--mouse-y', `${y}px`);
            };

            const handleTriggerClick = (e) => {
                if (!isExpanded) return;

                const triggerZone = e.currentTarget;
                const rect = triggerZone.getBoundingClientRect();
                const x = e.clientX - rect.left + triggerZone.offsetLeft;
                const y = e.clientY - rect.top + triggerZone.offsetTop;

                const text = CHAR_DIALOGUES.length > 0
                    ? CHAR_DIALOGUES[quoteIndex % CHAR_DIALOGUES.length]
                    : "点击空白区域触发对话";

                if (CHAR_DIALOGUES.length > 0) {
                    setQuoteIndex((prev) => (prev + 1) % CHAR_DIALOGUES.length);
}


                const bubbleId = Date.now();
                const newBubble = {
                    id: bubbleId,
                    x: x,
                    y: y,
                    text: text,
                    isFading: false
                };

                setBubbles(prev => {
                    const newQueue = [...prev, newBubble];
                    if (newQueue.length > 2) {
                        return newQueue.slice(newQueue.length - 2);
                    }
                    return newQueue;
                });

                setTimeout(() => {
                    setBubbles(prev => prev.map(b => b.id === bubbleId ? { ...b, isFading: true } : b));
                }, 5000);

                setTimeout(() => {
                    setBubbles(prev => prev.filter(b => b.id !== bubbleId));
                }, 6500);
            };

            const handleImageUpload = (e) => {
                const file = e.target.files[0];
                if (!file) return;

                const reader = new FileReader();
                reader.onloadend = () => {
                    const base64Result = reader.result;
                    setBgUrl(base64Result);

                    try {
                        localStorage.setItem('dsi_bg', base64Result);
                        setBgSaveStatus('saved');
                    } catch (error) {
                        console.warn('localStorage quota exceeded. Image will only be available for this session.', error);
                        setBgSaveStatus('session-only');
                        alert('图片较大，无法永久保存。背景将在本次会话中生效，刷新页面后会恢复默认。');
                    }
                };
                reader.onerror = () => {
                    console.error('Failed to read file.');
                    alert('读取文件失败，请重试。');
                };
                reader.readAsDataURL(file);
                e.target.value = null;
            };

            // --- 解析搜索历史 ---
            const searchRecords = parseSearchHistory(`$11`);

            const details = [
                {
                    id: 'loc',
                    icon: <i className="fa-solid fa-map-location-dot"></i>,
                    title: 'LOCATION / TIME',
                    content: (
                        <div className="info-stack">
                            <div className="detail-row"><span className="detail-label">Coordinates</span><span className="detail-value">$2</span></div>
                            <div className="detail-row" style={{ borderBottom: 'none' }}><span className="detail-label">Local Time</span><span className="detail-value text-bold">$3</span></div>
                            <div className="text-desc">$4</div>
                        </div>
                    )
                },
                {
                    id: 'outfit',
                    icon: <i className="fa-solid fa-user-astronaut"></i>,
                    title: 'OUTFIT / POSE',
                    content: (
                        <div className="tag-grid">
                            <div className="tag-box"><span className="detail-label mb-1">Upper</span>$5</div>
                            <div className="tag-box"><span className="detail-label mb-1">Lower</span>$6</div>
                            <div className="tag-box span-2"><span className="detail-label mb-1">Pose</span>$7</div>
                        </div>
                    )
                },
                {
                    id: 'mood',
                    icon: <i className="fa-solid fa-heart"></i>,
                    title: 'CURRENT MOOD',
                    content: (
                        <div className="mood-box">
                            <div className="mood-circle">$8</div>
                            <div className="mood-info">
                                <span className="text-bold" style={{ fontSize: '0.875rem' }}>$9</span>
                                <div className="text-desc" style={{ marginTop: '0.25rem', opacity: 0.7 }}>
                                    $10
                                </div>
                            </div>
                        </div>
                    )
                },
                {
                    id: 'search',
                    icon: <i className="fa-solid fa-magnifying-glass"></i>,
                    title: 'PRIVATE SEARCH',
                    content: (
                        <div className="search-list">
                            <div className="detail-row search-header"><span>QUERY</span><span>TIMESTAMP</span></div>
                            {searchRecords.length > 0 ? (
                                searchRecords.map((record, idx) => (
                                    <div key={idx} className="search-item">
                                        <span className="text-ellipsis">{record.content}</span>
                                        <span className="text-faint">{record.time}</span>
                                    </div>
                                ))
                            ) : (
                                <div className="text-desc" style={{textAlign: 'center', opacity: 0.5}}>暂无搜索记录</div>
                            )}
                        </div>
                    )
                },
                {
                    id: 'music',
                    icon: <i className="fa-solid fa-compact-disc"></i>,
                    title: 'AUDIO LOG',
                    content: <MusicPlayerUI
                                songData={songData}
                                isLoading={isLoadingMusic}
                                isPlaying={isPlaying}
                                progress={progress}
                                duration={duration}
                                volume={volume}
                                onTogglePlay={togglePlay}
                                onSeek={handleSeek}
                                onVolumeChange={handleVolume}
                             />
                },
                {
                    id: 'settings',
                    icon: <i className="fa-solid fa-gear"></i>,
                    title: 'SYSTEM SETTINGS',
                    content: (
                        <div className="info-stack">
                            <div className="detail-row" style={{ borderBottom: 'none', paddingBottom: 0 }}>
                                <span className="detail-label">Display Name</span>
                            </div>
                            <input
                                type="text"
                                className="setting-input"
                                value={charName}
                                onChange={(e) => setCharName(e.target.value)}
                                placeholder="Enter name..."
                            />

                            <div className="detail-row" style={{ borderBottom: 'none', paddingBottom: 0, marginTop: '8px' }}>
                                <span className="detail-label">Background Image</span>
                            </div>
                            <label className="file-upload-label">
                                <i className="fa-solid fa-cloud-arrow-up"></i>
                                <span>Upload (GIF/PNG/JPG)</span>
                                <input type="file" accept="image/png, image/jpeg, image/gif" hidden onChange={handleImageUpload} />
                            </label>
                            <div className="text-desc" style={{textAlign:'center', fontSize: '0.65rem'}}>
                                {bgSaveStatus === 'session-only'
                                    ? 'Image is too large. Only available this session.'
                                    : 'Image is saved locally in browser.'}
                            </div>
                        </div>
                    )
                }
            ];

            const activeItem = details.find(d => d.id === activeId);

            return (
                <div
                    onClick={() => !isExpanded && setIsExpanded(true)}
                    onMouseMove={handleMouseMove}
                    className={`swiss-island ${isExpanded ? 'expanded' : 'collapsed'}`}
                >
                    <audio ref={audioRef} style={{display:'none'}} />

                    <div className="island-bg-image" style={{ backgroundImage: `url(${bgUrl})`, opacity: isExpanded ? 1 : 0 }}></div>
                    <div className="island-overlay" style={{ opacity: isExpanded ? 1 : 0 }}></div>
                    {isExpanded && <div className="scan-line"></div>}

                    {isExpanded && bubbles.map(bubble => (
                        <div
                            key={bubble.id}
                            className={`speech-bubble ${bubble.isFading ? 'fading' : ''}`}
                            style={{ left: bubble.x, top: bubble.y - 20 }}
                        >
                            {bubble.text}
                        </div>
                    ))}

                    <div className="capsule-content">
                        <span className="capsule-title">{charName}</span>
                        <span className="capsule-date">$1</span>
                    </div>

                    <div className="card-content">

                        <div className="bubble-trigger-zone" onClick={handleTriggerClick}></div>

                        <div className="card-header-row">
                            <div>
                                <h1 className="main-title">{charName}</h1>
                                <div className="status-badge"><div className="status-dot"></div><span className="status-text">ACTIVE TRACKING</span></div>
                            </div>
                            <button onClick={(e) => { e.stopPropagation(); setIsExpanded(false); setActiveId(null); setBubbles([]); }} className="close-btn">
                                <svg width="14" height="14" viewBox="0 0 14 14" fill="none" stroke="currentColor" strokeWidth="2"><path d="M1 1L13 13M1 13L13 1" /></svg>
                            </button>
                        </div>

                        <div className="content-display-area" onClick={(e) => e.stopPropagation()}>
                            {activeItem && (
                                <div className="detail-glass-box" key={activeItem.id}>
                                    <div className="detail-header-title">{activeItem.title}</div>
                                    {activeItem.content}
                                </div>
                            )}
                        </div>

                        <div className="nav-footer" onClick={(e) => e.stopPropagation()}>
                            {details.map((item) => (
                                <button
                                    key={item.id}
                                    className={`nav-icon-btn ${activeId === item.id ? 'active' : ''}`}
                                    onClick={(e) => { e.stopPropagation(); setActiveId(activeId === item.id ? null : item.id); }}
                                >
                                    <span>{item.icon}</span>
                                    <div className="nav-dot"></div>
                                </button>
                            ))}
                        </div>
                    </div>
                </div>
            );
        };

        const root = ReactDOM.createRoot(document.getElementById('root'));
        root.render(<SwissDynamicIsland />);
    </script>
</body>

</html>
